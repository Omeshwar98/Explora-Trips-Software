-- üíΩ Create the Database
CREATE DATABASE IF NOT EXISTS exploratrips;
USE exploratrips;

-- üõ°Ô∏è ADMIN LOGIN TABLE
-- Stores login credentials for admin access
CREATE TABLE IF NOT EXISTS adminlogin (
    username VARCHAR(50),
    password VARCHAR(50)
);

-- Default admin login
INSERT INTO adminlogin (username, password) VALUES ('admin', '2025');

-- üßë‚Äçüíª ACCOUNT TABLE
-- Stores registered user accounts and their security information
CREATE TABLE IF NOT EXISTS account (
    username VARCHAR(50),
    name VARCHAR(50),
    password VARCHAR(50),
    security VARCHAR(50),
    answer VARCHAR(50)
);

-- üßë‚Äçüíº ADMIN TABLE
-- Stores additional admin details like contact and PAN info
CREATE TABLE IF NOT EXISTS admin (
    name VARCHAR(50),
    username VARCHAR(50),
    password VARCHAR(50),
    email VARCHAR(50),
    phone VARCHAR(50),
    pancard VARCHAR(50)
);

-- üßç CUSTOMER TABLE
-- Stores customer registration info including contact and identity details
CREATE TABLE IF NOT EXISTS customer (
    name VARCHAR(50),
    username VARCHAR(50),
    password VARCHAR(50),
    email VARCHAR(50),
    phone VARCHAR(50),
    pancard VARCHAR(50)
);

-- üè® BOOK HOTEL TABLE
-- Stores hotel booking details made by customers
CREATE TABLE IF NOT EXISTS bookhotel (
    hotelname VARCHAR(50),
    name VARCHAR(50),
    username VARCHAR(50),
    persons VARCHAR(50),
    days VARCHAR(50),
    ac_nonac VARCHAR(50),
    food VARCHAR(50),
    totalprice VARCHAR(50)
);

-- üé´ BOOK PACKAGE TABLE
-- Stores package bookings made by customers
CREATE TABLE IF NOT EXISTS bookpackage (
    place VARCHAR(50),
    name VARCHAR(50),
    username VARCHAR(50),
    persons VARCHAR(50),
    date VARCHAR(50),
    totalprice VARCHAR(50)
);

-- üè¢ HOTEL TABLE
-- Stores hotel listings, options, and image uploads
CREATE TABLE IF NOT EXISTS hotel (
    name VARCHAR(50),
    state VARCHAR(50),
    ac VARCHAR(50),
    food VARCHAR(50),
    hotelcost VARCHAR(50),
    image LONGBLOB
);

-- üß≥ PACKAGE TABLE
-- Stores tour packages available by state/place with price, description, and image
CREATE TABLE IF NOT EXISTS package (
    place VARCHAR(50),
    state VARCHAR(50),
    price VARCHAR(50),
    days_nights VARCHAR(50),
    description LONGTEXT,
    image LONGBLOB
);
